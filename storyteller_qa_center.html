import { test, expect } from '@playwright/test';

test.describe('Storyteller Web SDK Integration', () => {
  
  test.beforeEach(async ({ page }) => {
    await page.goto('https://example.com/stories');
  });

  test('should load Stories SDK', async ({ page }) => {
    // Verify SDK loaded
    const sdkLoaded = await page.evaluate(() => {
      return typeof window.Storyteller !== 'undefined';
    });
    
    expect(sdkLoaded).toBeTruthy();
  });

  test('should display story grid', async ({ page }) => {
    const storyGrid = page.locator('.storyteller-stories-grid');
    await expect(storyGrid).toBeVisible();
    
    const stories = page.locator('.storyteller-story-tile');
    await expect(stories).toHaveCount.toBeGreaterThan(0);
  });

  test('should open story on click', async ({ page }) => {
    const firstStory = page.locator('.storyteller-story-tile').first();
    await firstStory.click();
    
    const viewer = page.locator('.storyteller-story-viewer');
    await expect(viewer).toBeVisible();
  });

  test('should track analytics events', async ({ page }) => {
    let analyticsEvents = [];
    
    // Intercept analytics calls
    await page.route('**/api/analytics/**', route => {
      analyticsEvents.push(route.request().url());
      route.continue();
    });
    
    const story = page.locator('.storyteller-story-tile').first();
    await story.click();
    
    // Wait for analytics
    await page.waitForTimeout(2000);
    
    expect(analyticsEvents.length).toBeGreaterThan(0);
  });

  test('should handle responsive layout', async ({ page }) => {
    // Test mobile viewport
    await page.setViewportSize({ width: 375, height: 667 });
    
    const grid = page.locator('.storyteller-stories-grid');
    await expect(grid).toBeVisible();
    
    // Test tablet viewport
    await page.setViewportSize({ width: 768, height: 1024 });
    await expect(grid).toBeVisible();
    
    // Test desktop viewport
    await page.setViewportSize({ width: 1920, height: 1080 });
    await expect(grid).toBeVisible();
  });
});

import XCTest

class StoriesUITests: XCTestCase {
    
    var app: XCUIApplication!
    
    override func setUp() {
        super.setUp()
        app = XCUIApplication()
        app.launch()
    }
    
    func testLiveStoryDisplaysFirst() {
        let storiesGrid = app.otherElements["stories-grid"]
        XCTAssertTrue(storiesGrid.waitForExistence(timeout: 5))
        
        let firstStory = storiesGrid.cells.firstMatch
        let liveBadge = firstStory.staticTexts["LIVE"]
        
        XCTAssertTrue(liveBadge.exists)
    }
    
    func testStoryPageDuration() {
        let story = app.cells["story-card"].firstMatch
        story.tap()
        
        let startTime = Date()
        
        // Wait for auto-advance (15s default)
        let nextPageIndicator = app.otherElements["page-2"]
        XCTAssertTrue(nextPageIndicator.waitForExistence(timeout: 16))
        
        let duration = Date().timeIntervalSince(startTime)
        XCTAssertEqual(duration, 15, accuracy: 1)
    }
    
    func testEngagementUnitInteraction() {
        let story = app.cells["story-card"].firstMatch
        story.tap()
        
        // Navigate to poll page
        let pollOption = app.buttons["poll-option-1"]
        XCTAssertTrue(pollOption.waitForExistence(timeout: 5))
        
        pollOption.tap()
        
        let confirmation = app.staticTexts["Thanks for voting!"]
        XCTAssertTrue(confirmation.exists)
    }
    
    func testShareFunctionality() {
        let story = app.cells["story-card"].firstMatch
        story.tap()
        
        let shareButton = app.buttons["share-button"]
        shareButton.tap()
        
        let shareSheet = app.sheets.firstMatch
        XCTAssertTrue(shareSheet.waitForExistence(timeout: 3))
    }
}

          --format=junit \
          --output=test-results/ios
    
    - name: Run Android Tests
      if: matrix.platform == 'android'
      run: |
        maestro test flows/story_navigation.yaml \
          --device="Pixel 8" \
          --format=junit \
          --output=test-results/android

      with:
