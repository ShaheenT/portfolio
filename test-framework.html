import { Page, Locator } from '@playwright/test';

export class BasePage {
  protected page: Page;

  constructor(page: Page) {
    this.page = page;
  }

  // Navigate to a URL
  async goto(url: string) {
    await this.page.goto(url, { 
      waitUntil: 'networkidle' 
    });
  }

  // Wait for element to be visible
  async waitForElement(locator: Locator) {
    await locator.waitFor({ state: 'visible' });
  }

  // Click with retry logic
  async clickElement(locator: Locator) {
    await this.waitForElement(locator);
    await locator.click();
  }

  // Type with delay for realistic input
  async typeText(locator: Locator, text: string) {
    await this.waitForElement(locator);
    await locator.fill('');
    await locator.type(text, { delay: 50 });
  }

  // Get text content
  async getText(locator: Locator): Promise<string> {
    await this.waitForElement(locator);
    return await locator.textContent() || '';
  }

  // Take screenshot with custom name
  async takeScreenshot(name: string) {
    await this.page.screenshot({ 
      path: `screenshots/${name}.png`,
      fullPage: true
    });
  }

  // Wait for API response
  async waitForApiResponse(urlPattern: string) {
    return await this.page.waitForResponse(
      response => response.url().includes(urlPattern)
    );
  }
}

class StoriesScreen {
  // Selectors
  get storiesTab() {
    return $('~Stories'); // Accessibility ID
  }
  
  get storiesGrid() {
    return $('~stories-grid');
  }
  
  get storyCards() {
    return $('~story-card');
  }
  
  get storyViewer() {
    return $('~story-viewer');
  }
  
  get closeButton() {
    return $('~close-button');
  }
  
  // Actions
  async navigateToStories() {
    await this.storiesTab.waitForDisplayed({ timeout: 10000 });
    await this.storiesTab.click();
    await this.storiesGrid.waitForDisplayed();
  }
  
  async getStoryCount() {
    await this.storyCards[0].waitForDisplayed();
    return await this.storyCards.length;
  }
  
  async openStory(index = 0) {
    const cards = await this.storyCards;
    await cards[index].click();
    await this.storyViewer.waitForDisplayed();
  }
  
  async swipeNext() {
    const { width, height } = await driver.getWindowSize();
    await driver.touchPerform([
      { action: 'press', options: { x: width * 0.8, y: height * 0.5 }},
      { action: 'wait', options: { ms: 100 }},
      { action: 'moveTo', options: { x: width * 0.2, y: height * 0.5 }},
      { action: 'release' }
    ]);
  }
  
  async closeStoryViewer() {
    await this.closeButton.click();
    await this.storiesGrid.waitForDisplayed();
  }
  
  async takeScreenshot(name) {
    await driver.saveScreenshot(`./screenshots/${name}.png`);
  }
}

module.exports = new StoriesScreen();

STORY_SCHEMA = {
    "type": "object",
    "properties": {
        "id": {"type": "string"},
        "title": {"type": "string"},
        "media_url": {"type": "string"},
        "duration": {"type": "integer"}
    },
    "required": ["id", "title", "media_url"]
}

@pytest.mark.smoke
class TestStoriesAPI:
    
    def test_get_all_stories(self, api_client):
        """Test retrieving all stories"""
        response = api_client.get("/stories")
        
        assert response.status_code == 200
        assert "stories" in response.json()
        assert isinstance(response.json()["stories"], list)
        assert len(response.json()["stories"]) > 0
    
    def test_get_story_by_id(self, api_client):
        """Test retrieving a specific story"""
        # First get all stories
        stories = api_client.get("/stories").json()["stories"]
        story_id = stories[0]["id"]
        
        # Get specific story
        response = api_client.get(f"/stories/{story_id}")
        
        assert response.status_code == 200
        story = response.json()
        assert story["id"] == story_id
        
        # Validate schema
        validate(instance=story, schema=STORY_SCHEMA)
    
    def test_create_story(self, api_client, test_story_data):
        """Test creating a new story"""
        response = api_client.post("/stories", json=test_story_data)
        
        assert response.status_code == 201
        created_story = response.json()
        assert "id" in created_story
        assert created_story["title"] == test_story_data["title"]
    
    def test_update_story(self, api_client, test_story_data):
        """Test updating an existing story"""
        # Create a story first
        create_response = api_client.post("/stories", json=test_story_data)
        story_id = create_response.json()["id"]
        
        # Update the story
        updated_data = {"title": "Updated Story Title"}
        response = api_client.put(f"/stories/{story_id}", json=updated_data)
        
        assert response.status_code == 200
        assert response.json()["title"] == updated_data["title"]
    
    def test_delete_story(self, api_client, test_story_data):

        story_id = create_response.json()["id"]

        assert response.status_code == 204
        
        # Verify it's deleted
        get_response = api_client.get(f"/stories/{story_id}")
        assert get_response.status_code == 404
